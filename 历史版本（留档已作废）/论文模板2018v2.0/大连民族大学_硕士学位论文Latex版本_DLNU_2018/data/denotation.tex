\begin{denotation}

\item[$X$] 人脸特征数据集
\item[$X_i$] 第$i$个样本人脸特征
\item[$\mathbb{R}^{d}$] 实数空间的$d$维域
%\item[$Y$] 人脸的低维流形子空间
\item[$d(x_i,x_j)$] 表示样本$x_i$与$x_j$间的欧式距离
\item[$I$] 表示单位矩阵
\item[$coe$] 偏差系数
\item[$\otimes$] 卷积运算
\item[$G$] 生成式对抗网络的生成网络
\item[$D$] 生成式对抗网络的判别网络
\item[$m_{i,j}$] 描述第$i$个人脸特征的第$j$种语义概念


\end{denotation}
